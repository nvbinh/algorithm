function memoize(fun) {
	var result = null;
	return (arg) => {
		if(!result) {
			result = func(arg);
		}
		return result;
	}
}

function add(a) {
	return a + 1;
}

var memFun = memoize(add);

memFun(1) // Calculate
memFun(2) // not calculate again
